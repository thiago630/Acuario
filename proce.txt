CREATE PROCEDURE ActualizaSensores

(@ID_SENSORES INT,@NOMBRE_SEN VARCHAR(50),@MODELO_SEN VARCHAR(50),@MARCA_SEN VARCHAR(50),@COLOR_SEN VARCHAR(50),@MADEIN_SEN VARCHAR(50),@DETALLE_SEN VARCHAR(50),@STOCK_SEN INT,@FECHA DATE)


AS

/*Actualiza los sensores*/

IF NOT EXISTS (SELECT ID_SENSORES FROM sensores WHERE ID_SENSORES=@ID_SENSORES)

INSERT INTO sensores (ID_SENSORES,NOMBRE_SEN,MODELO_SEN,MARCA_SEN,COLOR_SEN,MADEIN_SEN,DETALLE_SEN,STOCK_SEN,FECHA) 

VALUES (@ID_SENSORES,@NOMBRE_SEN,@MODELO_SEN,@MARCA_SEN,@COLOR_SEN ,@MADEIN_SEN,@DETALLE_SEN,@STOCK_SEN,@FECHA)

ELSE
UPDATE sensores SET ID_SENSORES=@ID_SENSORES,NOMBRE_SEN=@NOMBRE_SEN,MODELO_SEN=@MODELO_SEN,MARCA_SEN=@MARCA_SEN,COLOR_SEN=@COLOR_SEN ,MADEIN_SEN=@MADEIN_SEN,DETALLE_SEN=@DETALLE_SEN,STOCK_SEN=@STOCK_SEN,FECHA=@FECHA
