CREATE DATABASE BOLTYCOM2;
USE BOLTYCOM2;

CREATE TABLE USUARIOS
(
	ID_USUARIO INT PRIMARY KEY AUTO_INCREMENT
	,NOM_USU VARCHAR(50)
	,ACCOUNT VARCHAR(50)
	,PASSWORD VARCHAR(50)
	,STATUS_ADMIN BIT
	,FOTO VARCHAR(500)


);
SELECT * FROM usuarios;

CREATE TABLE SENSORES
(
	ID_SENSORES INT PRIMARY KEY AUTO_INCREMENT
	,NOMBRE_SEN VARCHAR(50)
	,MODELO_SEN VARCHAR(50)
	,MARCA_SEN VARCHAR(50)
	,COLOR_SEN VARCHAR(50)
	,MADEIN_SEN VARCHAR(50)
	,DETALLE_SEN VARCHAR(50)
	,STOCK_SEN INT
	,FECHA DATE


);
